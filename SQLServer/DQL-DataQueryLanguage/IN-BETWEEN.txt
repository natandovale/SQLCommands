=>o 'in' e 'between) espera um array(valores), e executa a query entre esse array.
SELECT TOP 100 *
FROM [Curso]
WHERE [Id] IN (1,3)

SELECT TOP 100 *
FROM [Curso]
WHERE [Id] IN (1,2,3)

SELECT TOP 100 *
FROM [Curso]
WHERE [Id] BETWEEN 1 AND 3

SELECT TOP 100 *
FROM [Curso]
WHERE [Id] BETWEEN '2020-03-01 00:00:00' AND '2020-03-31 23:59:00'
